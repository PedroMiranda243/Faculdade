create type productType as object(
    id number,
    name varchar2(15),
    description varchar2(22),
    price number(5, 2),
    days_valid number
)

create table products(
    product productType,
    estoque number
)

select p.product.id, p.product.name, p.product.description, p.product.price, p.product.days_valid, estoque
from products p;
